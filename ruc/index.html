    RUC¶

    Consulta de RUC.

    Fuente: SUNAT.

    Ejemplo¶

    use Peru\Sunat\RucFactory;

require 'vendor/autoload.php';

$ruc = '20100070970';

$factory = new RucFactory();
$cs = $factory->create();

$company = $cs->get($ruc);
if (!$company) {
    echo 'Not found';
    return;
}

echo json_encode($company);
